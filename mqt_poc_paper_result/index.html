def load_sp500_returns(data_dir: str) -> tuple:
    """
    Load S&P 500 JSON files.  Each file: {"date": [...], "close": [...]}.
    Returns log-return matrix R (T x N) and ticker list.
    """
    files   = sorted(f for f in os.listdir(data_dir) if f.endswith(".json"))
    tickers = [f.replace(".json", "") for f in files]
    prices  = {}
    for f, ticker in zip(files, tickers):
        with open(os.path.join(data_dir, f)) as fh:
            d = json.load(fh)
        s = pd.Series(d["close"], index=pd.to_datetime(d["date"]), name=ticker)
        prices[ticker] = s

    df      = pd.DataFrame(prices).sort_index().ffill()
    returns = df.pct_change().dropna()
    return returns, tickers

# =============================================================================

def cvar_gaussian(mu_p: float, sigma_p: float, alpha: float) -> float:
    """Closed-form CVaR for Gaussian returns."""
    phi_inv = norm.ppf(alpha)
    return -mu_p + (sigma_p / (1 - alpha)) * norm.pdf(phi_inv)


def multilevel_cvar(weights: np.ndarray, mu: np.ndarray,
                    Sigma: np.ndarray) -> float:
    """Weighted multi-level CVaR across alpha levels (0.95, 0.99, 0.999)."""
    mu_p    = float(weights @ mu)
    sigma_p = float(np.sqrt(weights @ Sigma @ weights + 1e-10))
    total   = 0.0
    for alpha, w in zip(ALPHA_LEVELS, ALPHA_WEIGHTS):
        total += w * cvar_gaussian(mu_p, sigma_p, alpha)
    return CVaR_BETA * total


def dro_cvar(weights: np.ndarray, mu: np.ndarray,
             Sigma: np.ndarray, rng: np.random.Generator) -> float:
    """
    Distributionally Robust CVaR: worst-case over DRO_SAMPLES Gaussian
    perturbations of mu and Sigma within radius DRO_RADIUS.
    """
    worst = -np.inf
    for _ in range(DRO_SAMPLES):
        delta_mu    = rng.normal(0, DRO_RADIUS, size=mu.shape)
        delta_Sigma = rng.normal(0, DRO_RADIUS, size=Sigma.shape)
        delta_Sigma = 0.5 * (delta_Sigma + delta_Sigma.T)
        mu_p    = mu    + delta_mu
        Sigma_p = Sigma + delta_Sigma
        eigvals = np.linalg.eigvalsh(Sigma_p)
        if eigvals.min() < 0:
            Sigma_p -= (eigvals.min() - 1e-6) * np.eye(len(mu))
        val = multilevel_cvar(weights, mu_p, Sigma_p)
        worst = max(worst, val)
    return worst

# 3. HAMILTONIAN CONSTRUCTION  (QUBO → Ising → Pauli)
# =============================================================================

def build_hamiltonian(mu: np.ndarray, Sigma: np.ndarray,
                      budget: int, q: float = Q,
                      lam_b: float = PENALTY_B,
                      lam_l: float = PENALTY_L) -> SparsePauliOp:
    """
    Maps portfolio QUBO to Ising Hamiltonian via z_i = (1 - Z_i) / 2.
    H = H_risk + gamma*H_return + H_constraint
    """
    n = len(mu)
    pauli_list = []

    for i in range(n):
        for j in range(i, n):
            coeff = Sigma[i, j] if i == j else 2 * Sigma[i, j]
            if abs(coeff) < 1e-10:
                continue
            if i == j:
                op_str = ["I"] * n; op_str[i] = "Z"
                pauli_list.append(("".join(reversed(op_str)), -coeff / 4))
                pauli_list.append(("I" * n, coeff / 4))
            else:
                op_ij = ["I"] * n; op_ij[i] = "Z"; op_ij[j] = "Z"
                pauli_list.append(("".join(reversed(op_ij)), coeff / 4))
                op_i = ["I"] * n; op_i[i] = "Z"
                pauli_list.append(("".join(reversed(op_i)), -coeff / 4))
                op_j = ["I"] * n; op_j[j] = "Z"
                pauli_list.append(("".join(reversed(op_j)), -coeff / 4))
                pauli_list.append(("I" * n, coeff / 4))

    for i in range(n):
        if abs(mu[i]) < 1e-10:
            continue
        op_str = ["I"] * n; op_str[i] = "Z"
        pauli_list.append(("".join(reversed(op_str)), q * mu[i] / 2))
        pauli_list.append(("I" * n, -q * mu[i] / 2))

    for i in range(n):
        op_str = ["I"] * n; op_str[i] = "Z"
        pauli_list.append(("".join(reversed(op_str)), -lam_b / 2))
        pauli_list.append(("I" * n, lam_b / 2 - lam_b * budget))
        for j in range(i + 1, n):
            op_ij = ["I"] * n; op_ij[i] = "Z"; op_ij[j] = "Z"
            pauli_list.append(("".join(reversed(op_ij)), lam_b / 2))
            op_i = ["I"] * n; op_i[i] = "Z"
            pauli_list.append(("".join(reversed(op_i)), -lam_b / 2))
            op_j = ["I"] * n; op_j[j] = "Z"
            pauli_list.append(("".join(reversed(op_j)), -lam_b / 2))
            pauli_list.append(("I" * n, lam_b / 2))

    if not pauli_list:
        pauli_list = [("I" * n, 0.0)]
    return SparsePauliOp.from_list(pauli_list).simplify()

def evaluate_energy(params: np.ndarray, ansatz, hamiltonian,
                    estimator) -> float:
    """
    Compute <psi(theta)|H|psi(theta)> via the given Estimator primitive.
    Handles both Primitives V1 (AerEstimator) and V2 (StatevectorEstimator).
    """
    bound = ansatz.assign_parameters(params)
    try:
        # V2 API: StatevectorEstimator (qiskit.primitives)
        job = estimator.run([(bound, hamiltonian)])
        return float(job.result()[0].data.evs)
    except TypeError:
        # V1 API: AerEstimator (qiskit_aer.primitives)
        job = estimator.run([bound], [hamiltonian])
        return float(job.result().values[0])

# 6. ADAM OPTIMIZER WITH NOISE-TRIGGERED CMA-ES REVERT
# =============================================================================

def adam_optimize(params0: np.ndarray, loss_fn, rng: np.random.Generator,
                  history: list) -> np.ndarray:
    """
    Phase 2: ADAM with finite-difference gradients.
    Noise-triggered CMA-ES reverts when gradient variance exceeds threshold.
    """
    params  = params0.copy()
    m       = np.zeros_like(params)
    v       = np.zeros_like(params)
    best_p  = params.copy()
    best_v  = loss_fn(params)
    reverts = 0

    for t in range(1, ADAM_ITERS + 1):
        grads = np.zeros_like(params)
        for k in range(len(params)):
            p_plus  = params.copy(); p_plus[k]  += ADAM_DELTA
            p_minus = params.copy(); p_minus[k] -= ADAM_DELTA
            grads[k] = (loss_fn(p_plus) - loss_fn(p_minus)) / (2 * ADAM_DELTA)

        if t > REVERT_START and reverts < REVERT_MAX:
            grad_var = float(np.var(grads))
            if grad_var > GRAD_VAR_THR:
                bounds = [(-np.pi, np.pi)] * len(params)
                res    = dual_annealing(loss_fn, bounds,
                                        maxiter=CMAES_BUDGET,
                                        x0=best_p, seed=rng.integers(1000))
                if res.fun < best_v:
                    params  = res.x
                    best_v  = res.fun
                    best_p  = params.copy()
                reverts += 1

        m = ADAM_B1 * m + (1 - ADAM_B1) * grads
        v = ADAM_B2 * v + (1 - ADAM_B2) * grads ** 2
        m_hat = m / (1 - ADAM_B1 ** t)
        v_hat = v / (1 - ADAM_B2 ** t)
        params = params - ADAM_LR * m_hat / (np.sqrt(v_hat) + ADAM_EPS)

        current = loss_fn(params)
        history.append(current)
        if current < best_v:
            best_v  = current
            best_p  = params.copy()

    return best_p

print("adam_optimize() defined.")

def make_objective(ansatz, hamiltonian, estimator,
                   mu: np.ndarray, Sigma: np.ndarray,
                   budget: int, rng: np.random.Generator):
    """
    Joint objective:
      L(theta) = DRO-CVaR(x(theta)) + lambda * <H>(theta)
    """
    lam = 1.0

    def objective(params: np.ndarray) -> float:
        energy  = evaluate_energy(params, ansatz, hamiltonian, estimator)
        raw     = np.abs(np.sin(params[:len(mu)]))
        weights = raw / (raw.sum() + 1e-10)
        dro     = dro_cvar(weights, mu, Sigma, rng)
        pen_b   = PENALTY_B * (raw.sum() - budget) ** 2
        pen_l   = PENALTY_L * np.sum(np.clip(-raw, 0, None))
        return float(energy + lam * dro + pen_b + pen_l)

    return objective

from scipy.optimize import minimize as sp_minimize

def run_experiment(data_dir: str = "./data/sp500",
                   seed: int = 42,
                   noisy: bool = False) -> dict:
    """
    Full DRO-VQE experiment.
    Phase 1: Dual Annealing (global search, capped at maxfun=100)
    Phase 2: COBYLA (gradient-free, ~100 evals — replaces ADAM finite-diff
             which needed 2×n_params evals per step = 450+ evals for 15 params)
    """
    rng = np.random.default_rng(seed)

    returns, tickers = load_sp500_returns(data_dir)
    mu     = returns.mean().values
    Sigma  = returns.cov().values
    n      = len(mu)
    budget = n // 2

    print(f"Assets: {n}  |  Budget: {budget}  |  Noisy: {noisy}")
    print(f"Tickers : {tickers}")
    print(f"Returns : shape={returns.shape}, mean={mu.mean():.5f}, std={mu.std():.5f}")

    hamiltonian = build_hamiltonian(mu, Sigma, budget)
    ansatz      = build_ansatz(n)
    n_params    = ansatz.num_parameters
    print(f"Hamiltonian : {len(hamiltonian)} Pauli terms  |  Ansatz params: {n_params}")

    if noisy:
        noise_model = NoiseModel()
        noise_model.add_all_qubit_quantum_error(depolarizing_error(0.001, 1), ["ry"])
        noise_model.add_all_qubit_quantum_error(depolarizing_error(0.01,  2), ["cz"])
        estimator = AerEstimator(
            backend_options={"noise_model": noise_model},
            run_options={"shots": SHOTS_NOISY})
        print(f"Backend : AerEstimator  (depolarising noise, shots={SHOTS_NOISY})")
    else:
        estimator = StatevectorEstimator()
        print(f"Backend : StatevectorEstimator (exact, shots=None)")

    objective = make_objective(ansatz, hamiltonian, estimator,
                               mu, Sigma, budget, rng)
    history   = []

    def _tracked(p):
        val = objective(p)
        history.append(val)
        return val

    # ── Phase 1: Dual Annealing (global exploration) ──────────────────────────
    print(f"\nPhase 1: Dual Annealing  (maxfun=100 hard cap) ...")
    bounds  = [(-np.pi, np.pi)] * n_params
    da_res  = dual_annealing(
        _tracked, bounds,
        maxiter=DA_MAXITER, maxfun=100,
        seed=int(rng.integers(10000)),
    )
    best_params = da_res.x
    print(f"  Done — best energy: {da_res.fun:.4f}  |  evals: {len(history)}")

    # adaptive ansatz depth check
    if len(history) > 10 and np.var(history[-10:]) < STAGNATION_THRESH:
        new_reps = min(ansatz.reps + 1, 3)
        if new_reps != ansatz.reps:
            print(f"  Stagnation → increasing reps to {new_reps}")
            ansatz      = build_ansatz(n, reps=new_reps)
            n_params    = ansatz.num_parameters
            objective   = make_objective(ansatz, hamiltonian, estimator,
                                         mu, Sigma, budget, rng)
            best_params = np.concatenate([
                best_params,
                rng.uniform(-np.pi, np.pi, n_params - len(best_params))
            ])

    # ── Phase 2: COBYLA (gradient-free refinement) ────────────────────────────
    # COBYLA needs ~100 total evals vs ADAM's 2×n_params×iters = 450 evals
    print(f"\nPhase 2: COBYLA refinement  (maxiter=120) ...")
    cobyla_res  = sp_minimize(
        _tracked, best_params,
        method='COBYLA',
        options={'maxiter': 120, 'rhobeg': 0.3},
    )
    best_params = cobyla_res.x
    final_obj   = _tracked(best_params)
    print(f"  Done — final objective: {final_obj:.4f}  |  total evals: {len(history)}")

    # decode weights and metrics
    raw     = np.abs(np.sin(best_params[:n]))
    weights = raw / (raw.sum() + 1e-10)
    mu_p    = float(weights @ mu)
    sigma_p = float(np.sqrt(weights @ Sigma @ weights + 1e-10))
    cvar_v  = multilevel_cvar(weights, mu, Sigma)
    sharpe  = mu_p / (sigma_p + 1e-10)

    print(f"\n{'='*55}")
    print(f"  Annualised Return  : {mu_p * 252:.4f}")
    print(f"  Volatility (ann.)  : {sigma_p * np.sqrt(252):.4f}")
    print(f"  Sharpe Ratio       : {sharpe:.4f}")
    print(f"  Multi-level CVaR   : {cvar_v:.6f}")
    print(f"{'='*55}")
    print("Portfolio weights:")
    for idx in np.argsort(weights)[::-1]:
        bar = '█' * int(weights[idx] * 40)
        print(f"  {tickers[idx]:<8}  {weights[idx]:.4f}  {bar}")

    return {
        "weights": weights, "tickers": tickers,
        "mu": mu, "Sigma": Sigma,
        "return": mu_p, "cvar": cvar_v, "sharpe": sharpe,
        "history": history, "noisy": noisy,
    }
